% Räknare för frågor
\newcounter{fraga}
\newcounter{fragaonpage} % håller koll på hur många frågor på aktuell sida

% Anpassa rubrikstil
\titleformat{\section}{\centering\bfseries\Large}{}{0pt}{}

% Kommando för en fråga
% Argument: #1 = bildfil (kan vara tom), #2 = frågetext, #3 = svar 1, #4 = svar X, #5 = svar 2
\newcommand{\tipspromenadfraga}[5]{%
  \stepcounter{fraga}%
  \stepcounter{fragaonpage}%
  % Om det är första frågan på sidan -> börja en rad
  \ifnum\value{fragaonpage}=1
    \noindent
  \fi
  % Frågeruta i en minipage
  \begin{minipage}[t]{0.48\textwidth}
    \begin{center}
      \textbf{\LARGE TIPSPROMENAD}\\[1em]
      \textbf{\Large FRÅGA \thefraga}
    \end{center}
    \vspace{1em}
    % Bild (om angiven)
    \ifstrempty{#1}{}{
      \begin{center}
        \includegraphics[height=4cm,keepaspectratio]{#1}
      \end{center}
    }
    % Frågetext
    \begin{center}
      \parbox{13cm}{\centering #2}
    \end{center}
    \vspace{1em}
    % Svarsalternativ
    \renewcommand{\arraystretch}{1.8}
    \begin{center}
      \begin{tabular}{|>{\centering\arraybackslash}m{1cm}|m{11cm}|}
        \hline
        1 & #3 \\ \hline
        X & #4 \\ \hline
        2 & #5 \\ \hline
      \end{tabular}
    \end{center}
  \end{minipage}%
  % Om det är andra frågan på sidan -> radslut + sidbrytning
  \ifnum\value{fragaonpage}=2
    \setcounter{fragaonpage}{0}%
    \par\newpage
  \else
    \hfill
  \fi
}

\begin{document}

\tipspromenadfraga{stockholm.jpg}{Vilken stad är Sveriges huvudstad?}
{Stockholm}{Göteborg}{Malmö}

\tipspromenadfraga{}{Hur många grader är en rät vinkel?}
{45}{90}{180}

\tipspromenadfraga{moon.jpg}{Vilket år landade människan på månen?}
{1965}{1969}{1972}

\tipspromenadfraga{}{Vilket grundämne har kemiska beteckningen O?}
{Guld}{Syre}{Silver}

\end{document}
